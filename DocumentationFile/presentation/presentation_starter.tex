\section{Introduction}
\frameT{Motivation} {
  \bigskip
  \begin{itemize}
  	\raggedright
    \item To make a Classical FPS shooter game with inspiration from the game Quake as there have been a lower amount of arena shooters compared to the past couple decades.
      \bigskip
    \item The idea of the game is to have players be rodents like mice fighting each other with different kinds of weapons and jumping around the map.
  \end{itemize}
  
  	\begin{center}
  		\includegraphics[width=.8\linewidth]{figures/QuakeII.jpg}
  	\end{center}

  \bigskip
  
  \emph{ 
} \emph{ }.
}


\frameT{Technology} {
  Technology used:
  \bigskip
  \begin{itemize}
    \item Unreal Engine 5: Used for the creation of the overall game.
    \begin{itemize}
      \item Ultimate Doom Builder: Used to create the maps used.
      \item Blender: Used to make the weapons used in the game along with helping import the maps.
    \end{itemize}
  \end{itemize}
  
    	\begin{center}
  		\includegraphics[width=.8\linewidth]{figures/LogoUsed.png}
  	\end{center}
  
  \bigskip
}


\frameT{Project Goals} {
  \begin{itemize}
      \item Networking testing to where players will be able to play together from different devices on the same server.
      \item Multiple weapons and Maps Available to the players.
      \item AI enemies available to substitute for players.
  \end{itemize}
  
    	\begin{center}
  		\includegraphics[width=.6\linewidth]{figures/TestDemoGame2.png}
  	\end{center}
  
  \bigskip
}

\section{Tools}

\frameT{Map Creation Pipeline} {
	\begin{center}
  		\includegraphics[width=.8\linewidth]{figures/map_pipeline.png}
  	\end{center}
  	
  	\begin{multicols}{3}
        \begin{itemize}
        		\small
            \item The maps are made in Ultimate Doom Builder (UDB), they are exported into a .obj file.
            \item We import the .obj file into Blender, where we exported from Blender as an FBX file.
            \item Finally, the FBX file can be imported to UE as a model.
        \end{itemize}
    \end{multicols}
}

\frameT{Difficulties}{
  Challenges:
  \begin{itemize}
      \item Networking has been a bit of a challenge to have work from different devices currently.
      \item Had minor spawning issues for weapons with certain maps.
      \item Sound not working on some of the weapons on certain maps.
  \end{itemize}
}

\frameT{Accomplishments}{
  Accomplished:
  \begin{itemize}
      \item The maps are working
      \item Base Game play is accomplished for player control, shooting, and hit registration.
      \item Different weapons and maps made.
      \item Progress made on the Network Tests.
  \end{itemize}
}

\frameT{Future Goals}{
  Goals:
  \begin{itemize}
      \item Fix the current issues we have difficulty with.
      \item Creation of the rodent character models.
      \item Work more on character respawning.
      \item Develop the enemy AIs.
  \end{itemize}
}
